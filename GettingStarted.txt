/*install WSL Ubuntu*/
 **Note**

 The project will not compile if the source is downloaded outside of WSL, and accessed via the /mnt/C folder structure.
 Instead, you need to download the git repo from within WSL (in your home directory, or wherever you prefer)

sudo apt-get update
sudo apt-get install python3 python3-pip git zlib1g-dev libopenmpi-dev ffmpeg python3-opengl
sudo pip3 install -U virtualenv
virtualenv --system-site-packages -p python3 ~/vretro
source ~/vretro/bin/activate
cd /home/marciolsf/marciolsf/stable-retro
pip3 install -e .


pip3 install stable_baselines3[extra]
